proc reg data=imports;
	model enginesize=wheelbase length curbweight hp citympg highwaympg/ selection=stepwise sle=.05 sls=.05;
	output out=diag cookd=cd;
	ods select SelectionSummary DiagnosticsPanel;
run;
proc reg data=diag;
	model enginesize=curbweight hp citympg;
	where cd< .5;
	output out=diag2 cookd=cd2;
	ods select DiagnosticsPanel;
run;
proc reg data=diag2;
	model enginesize=curbweight hp citympg;
	where cd2< .3;
	output out=diag3 cookd=cd3;
	ods select DiagnosticsPanel;
run;

proc reg data=diag3;
	model enginesize=curbweight hp citympg;
	where cd3< .275;
	output out=diag4 cookd=cd4;
	ods select DiagnosticsPanel;
run;

	model enginesize=curbweight hp citympg;
	where cd4< .3;
