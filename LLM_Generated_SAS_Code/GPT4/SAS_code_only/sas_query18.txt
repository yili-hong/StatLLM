PROC UNIVARIATE DATA=sleeptime;
  VAR Time;
  HISTOGRAM / NORMAL;
  INSET MEAN STDDEV MIN Q1 MEDIAN Q3 MAX/ FORMAT=5.2;
  PROBT;
  PROBN;
  QQPLOT /NORMAL (MU=EST SIGMA=EST);
RUN;

%LET MedianGuess = 7;

PROC UNIVARIATE DATA=sleeptime NOPRINT;
  VAR Time;
  OUTPUT OUT=Temp PCTLPTS= 50 PCTLPRE=P;
RUN;

DATA Temp;
  SET Temp;
  MEDIAN_H0 = &MedianGuess;
  DIFF = P50 - MEDIAN_H0;
RUN;

PROC UNIVARIATE DATA=Temp;
  VAR DIFF;
  HISTOGRAM / WINSOR=1;
RUN;
