
/* Step 2: Perform the Levene's Test using PROC GLM (General Linear Model Procedure) */

  CLASS Fertilizer;  /* categorical variable */

  MEANS Fertilizer / HOVTEST=LEVENE;  /* performs the Leveneâ€™s test */
