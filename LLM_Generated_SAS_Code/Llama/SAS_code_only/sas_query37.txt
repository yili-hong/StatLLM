/* Summary statistics for the dataset */
proc means data=marketing_data;
  var Opened_Email Clicked_Link Purchased Revenue;
run;

/* Frequency distribution for categorical variables */
proc freq data=marketing_data;
  tables Group;
run;

proc logistic data=marketing_data;
  model Purchased(event='1') = Group Opened_Email Clicked_Link / link=logit;
  oddsratio;
run;

/* Decision tree analysis to identify key factors */
proc dtree data=marketing_data;
  var Group Opened_Email Clicked_Link;
  target Purchased;
  prune;
run;

proc logistic data=marketing_data;
  model Purchased(event='1') = Group Opened_Email Clicked_Link / link=logit;
  score data=marketing_data out=predicted;
  roc;
run;

proc print data=predicted;
  var Customer_ID Predicted;
run;

proc ttest data=marketing_data;
  class Group;
  var Revenue;
run;
